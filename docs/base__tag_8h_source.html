371 virtual inline size_t WriteToBuffer(char* buffer,
372 size_t length,
373 size_t offset) const {
374 if (offset + sizeof(uint32_t) > length) {
375 return NBTCC_WRITE_ERROR;
376 }
377
378 // First `sizeof(uint32_t)` bytes is the size of array.
379 const std::vector<T>* array = GetVector();
380 uint32_t size = array->size();
381 uint32_t size_be = htobe32(size);
382 memcpy(buffer + offset, &size_be, sizeof(uint32_t));
383
384 // Then `sizeof(T) * size` bytes is the array.
385 size_t next_offset = offset + sizeof(uint32_t);
386 size_t end_offset = next_offset + size * sizeof(T);
387
388 if (end_offset > length) {
389 return NBTCC_WRITE_ERROR;
390 }
391
392 memcpy(buffer + next_offset, array->data(), size * sizeof(T));
393 for (char* i = buffer + next_offset; i < buffer + end_offset;
394 i += sizeof(T)) {
395 if (sizeof(T) == 2) {
396 *reinterpret_cast<uint16_t*>(i) =
397 be16toh(*reinterpret_cast<uint16_t*>(i));
398 } else if (sizeof(T) == 4) {
399 *reinterpret_cast<uint32_t*>(i) =
400 be32toh(*reinterpret_cast<uint32_t*>(i));
401 } else if (sizeof(T) == 8) {
402 *reinterpret_cast<uint64_t*>(i) =
403 be64toh(*reinterpret_cast<uint64_t*>(i));
404 }
405 }
406
407 return size * sizeof(T) + sizeof(uint32_t);
408 }

431 array->resize(array_size);
432 T* data = array->data();
433 for (size_t i = next_offset, j = 0; i < end_offset; i += sizeof(T), j++) {
434 if (sizeof(T) == 2) {
435 *reinterpret_cast<uint16_t*>(data + j) =

446 }
